# Post-Quantum Cryptography Libraries

# Default production-grade demo backend:
# Provides ML-DSA-65 and SPHINCS+-SHA2-256f verification.
pqcrypto==0.4.0

# Optional alternatives:

# Option 1: Open Quantum Safe (Recommended) - Provides ML-DSA, ML-KEM, SLH-DSA
# Supports NIST-standardized post-quantum algorithms
# Installation: pip install liboqs-python
# liboqs-python>=0.7.2

# Option 2: SPHINCS+ - Hash-based stateless signatures
# Fast and secure, no state management needed
# Installation: pip install sphincsplus
# sphincsplus>=0.7.0

# Option 3: XMSS - eXtendable Merkle Signature Scheme
# Hash-based with smaller signatures than SPHINCS+
# No maintained pip package was available for this Python 3.14/Windows setup
# during project hardening. Keep this as an extension point unless using a
# known-good XMSS package or a pinned older Python/runtime.
# Historical installation note: pip install xmss-py
# xmss-py>=1.1.0

# If pqcrypto is unavailable on a platform, the legacy WOTS-SHA256 fallback
# remains available for local classroom demonstrations.

# Installation instructions:
# 1. Create virtual environment: python -m venv venv
# 2. Activate: venv\Scripts\activate (Windows) or source venv/bin/activate (Unix)
# 3. Install core + ML-DSA: pip install -r requirements.txt
# 4. Install optional alternatives only when needed.
